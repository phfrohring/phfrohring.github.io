<nav id="navigation" class="navigation">

JavaScript:

!(function(){
    'use strict'

window.onload = function() {
    var navigation_id = 'navigation'
    var locations_wrapper = 'locations-wrapper'

    var addEventListenerMulti = function(element, events, listner) {
        events.split(' ').forEach(evt => element.addEventListener(evt, listner));
    }

    var when_needed_set_locations_wrapper_width =
        function(navigation_id, locations_ids) {
            var navigation = document.getElementById(navigation_id)
            var navigation_w = window.getComputedStyle(navigation).width
            var locations_wrapper = document.getElementById(locations_ids);
            if (locations_wrapper.style.position === 'fixed') {
                locations_wrapper.style.width = `calc(${navigation_w} - 1em)`
                locations_wrapper.style.marginLeft = '1em'
            }
            else {
                locations_wrapper.style.width = 'auto'
                locations_wrapper.style.marginLeft = '1em'
        }
    }

    var when_needed_set_locations_wrapper_y =
        function(navigation_id, locations_ids) {
            var navigation = document.getElementById(navigation_id)
            var navigation_w = window.getComputedStyle(navigation).width
            var navigation_y = navigation.offsetTop;
            var locations_wrapper = document.getElementById(locations_ids);
            if (window.pageYOffset > navigation_y) {
                locations_wrapper.style.position = "fixed";
                locations_wrapper.style.top = "0";
            } else {
                locations_wrapper.style.position = "";
                locations_wrapper.style.top = "";
        }
    }


    when_needed_set_locations_wrapper_y(navigation_id, locations_wrapper)
    when_needed_set_locations_wrapper_width(navigation_id, locations_wrapper)


    addEventListenerMulti(window,'scroll resize orientationChange', function() {
        when_needed_set_locations_wrapper_y(navigation_id, locations_wrapper)
        when_needed_set_locations_wrapper_width(navigation_id, locations_wrapper)
    })
}

}())
